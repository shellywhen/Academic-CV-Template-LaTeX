\newcommand{\reviewSparkSpike}[2]{%
  {%
    \pgfmoveto{\pgforigin}%
    \pgfsetlinewidth{\sparkspikewidth}%
    \pgfsetstrokecolor{sparkspikecolor}%
    \pgfmoveto{\pgfxy(#1, 0)}%
    \pgflineto{\pgfxy(#1, #2)}%
    \pgfusepath{stroke}%
  }%
}

  \edef\reviewX{\fpeval{\reviewXMin + ((#1-1) * (\reviewXMax-\reviewXMin) / (\reviewPointCount-1))}}%
  \edef\reviewHeight{\fpeval{#2/\reviewCountMax}}%
  \reviewSparkSpike{\reviewX}{\reviewHeight}%
}
